import sys
from collections import deque
def find(width):
    global blue
    for i in range(width):
        for j in range(width):
            if not visited[i][j]:
                if rgb[i][j]=='B':
                    blue +=1
                queue.append([i,j])

def find_rg(width):
    for i in range(width):
        for j in range(width):
            if not visited[i][j]:
                if rgb[i][j]!='B':
                    queue.append([i,j])
                    return True
                

dx = [0,0,1,-1]
