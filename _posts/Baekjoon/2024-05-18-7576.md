    tomato.append(list(map(int, input().strip().split())))

    for j in range(m):

            queue.append([tomato[i][j],i,j,day])
        elif tomato[i][j]==-1:
            box +=1
        elif tomato[i][j] == 0:
            zero +=1
if zero>0:
    while queue:
        value, i,j,day=queue.popleft()
        if not visited[i][j] and value ==1:
            visited[i][j] = True
            ripen+=1
            for z in range(4):
                x = dx[z]+i
                y = dy[z]+j
                if 0<=x<n and 0<=y<m:

                    if tomato[x][y]!=-1:
                        tomato[x][y]=1
                        queue.append([tomato[x][y],x,y,day+1])
    if box+ripen == n*m:
        print(day-1)
    else:
        print(-1)
else:
    print(0)
