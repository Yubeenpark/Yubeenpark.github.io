import sys

import math

    street.append(list(map(int,input().strip().split())))

    for j in range(n):
        if street[i][j]==2:
            chicken.append((i,j))

            home.append((i,j))

if m != len(chicken):
    select = combinations(chicken,m)
else:
    select = [chicken]

for chi_store in select:
    dist = 0
    for h in home:  #ê° ì§‘ì„ ëŒë©´ì„œ ì¹˜í‚¨ ê±°ë¦¬ êµ¬í•˜ê¸°
