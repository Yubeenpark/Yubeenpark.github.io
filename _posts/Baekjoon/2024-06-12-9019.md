```python
import sys
from collections import deque
input= sys.stdin.readline
t = int(input())

n=0

def D():
    if 2*n>9999:
        return (2*n)%10000
    else:
        return 2*n
    
def S():
    if n==0:
        return 9999
    else:
        return n-1
    
def L():
    return n // 1000 + (n % 1000)*10

def R():                        
    return n // 10 + (n % 10) * 1000
func = [D,S,L,R]

for _ in range(t):
    a,b = map(int,input().strip().split(' '))
    visited=[0 for _ in range(10000)]
    queue = deque()
    queue.append((a,''))
    while queue:
        n,case= queue.popleft()
        if n == b:
            print(case)
            break   
        for i in range(4):
            case_n  = func[i]()
            if visited[case_n]==0:
                queue.append((case_n,case+str(func[i].__name__)))
                visited[case_n]=1
```
